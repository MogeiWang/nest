47  // compute using Taylor series, see GSL
48  // e^x-1 = x + x^2/2! + x^3/3! + ...
49  if ( x == 0 )
50  return 0;
51  if ( std::abs(x) > std::log(2.0) )
52  return std::exp(x) - 1;
53  else
54  {
55  double sum = x;

58

61  {

65  }
